<script>


//========================

// MQTT CONNECT

//========================

const client = mqtt.connect("wss://broker.hivemq.com:8884/mqtt");


client.on("connect", () => {

console.log("MQTT conectado");


client.subscribe("boat/data");

client.subscribe("boat/status");

});


//========================

// DATA HANDLER

//========================

client.on("message", (topic, message) => {


const msg = message.toString();


// STATUS

if(topic === "boat/status") {


const el = document.getElementById("status");


if(msg === "ONLINE") {

el.innerHTML = "ONLINE";

el.className = "status online";

} else {

el.innerHTML = "OFFLINE";

el.className = "status offline";

}


return;

}


// DATA JSON

try {

const d = JSON.parse(msg);


document.getElementById("ws").innerHTML = d.ws.toFixed(1);

document.getElementById("wd").innerHTML = d.wd.toFixed(1);

document.getElementById("depth").innerHTML = d.depth.toFixed(1);


document.getElementById("bat_m").innerHTML = d.bat_m.toFixed(2);

document.getElementById("bat_s").innerHTML = d.bat_s.toFixed(2);


document.getElementById("rssi").innerHTML = d.rssi;


} catch(e) {

console.log("JSON error", e);

}


});


</script>
